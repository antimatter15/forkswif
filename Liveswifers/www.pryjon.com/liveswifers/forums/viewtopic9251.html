just use the ajax animator to create a TWO FRAME ANIMATION
So just make a new animation, draw some stuff. and just go to the next frame (press right arrow key) and all the objects should be the same, and just move the objects around using the select thing.

Then,

go to the script executor (located in the advanced tab, on the bottom panel)

Paste:

Quote:
var e=(new DOMParser()).parseFromString(animationSaveData(),"text/xml").firstChild.getElementsByTagName("svg");
if(e[1].childNodes.length == e[0].childNodes.length){
var tweens = 50;// number of frames to tween
var newE = new Array();
newE[0] = e[0].cloneNode(true)
for(var ctf=0;ctf<tweens;ctf++){
newE[ctf] = e[0].cloneNode(true)}
for(var objIndex=0;objIndex<e[0].childNodes.length;objIndex++){
if(e[0].childNodes[objIndex].getAttribute("id") + e[1].childNodes[objIndex].getAttribute("id")){
var x1 = parseInt(e[0].childNodes[objIndex].getAttribute("x"))
var x2 = parseInt(e[1].childNodes[objIndex].getAttribute("x"))
var xtDistance = x2-x1;
var y1 = parseInt(e[0].childNodes[objIndex].getAttribute("y"))
var y2 = parseInt(e[1].childNodes[objIndex].getAttribute("y"))
var ytDistance = y2-y1;
var xtDfP = xtDistance/tweens;
var ytDfP = ytDistance/tweens;
for(var tf=0;tf<tweens;tf++){
newE[tf].childNodes[objIndex].setAttribute("x", (xtDfP * tf) + x1)
newE[tf].childNodes[objIndex].setAttribute("y", (ytDfP * tf) + y1)}}}
var aXMLSrc = "<AnimationXML>";
for(var cf=0;cf<newE.length;cf++){
aXMLSrc += (new XMLSerializer()).serializeToString(newE[cf]) }
aXMLSrc += "</AnimationXML>"
loadAnimation(aXMLSrc)}


press "Execute" and wait.

the two-frame animation will automatically be converted to a 50 frame animation (set the number of frames in the "tweens" variable, in the code above)
