SOLUTION:

onClipEvent (load)
Code:

      gravity = 5;
      speed = 5;
      isJump = true;



onClipEvent (enterFrame)
Code:
      ++gravity;
      _y += gravity;
      if (Key.isDown(Key.LEFT)) {
        _x -= speed;
      }
      if (Key.isDown(Key.RIGHT)) {
        _x += speed;
      }
      if (Key.isDown(Key.UP) && isJump == false) {
        isJump = true;
        _y -= 5;
        gravity = -13;
      }
      if (this.hitTest(_root.ground)) {
        _root.player.isJump = false;
        _root.player.gravity = 0;
        --_root.player._y;
      }
