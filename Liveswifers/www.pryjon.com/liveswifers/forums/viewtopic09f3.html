timeline.js wrote:
var layers = 0;
var kFrameCount = 0;
var currentFrameSelection = 1;
var currentLayerSelection = 1;
var KeyFrames = new Array()

function setOpacity(obj, value) {
getElementById(obj).style.opacity = value/10;
getElementById(obj).style.filter = 'alpha(opacity=' + value*10 + ')';
}

function toKeyframe()
{
var zframe;
zframe = document.getElementById("frame" + currentFrameSelection + "layer" + currentLayerSelection);
zframe.style.color = "#000000";
zframe.style.backgroundColor="#0099CC";
KeyFrames[kFrameCount] = currentFrameSelection + "," + currentLayerSelection
kFrameCount = kFrameCount + 1
currentFrameSelection = 1;
currentLayerSelection = 1;
}

function fullgotoframe(framenumber, layer)
{
if (confirm("Warning: This might take a very long time")) {
alert('you will be alerted when it is done');
for(var iz = 1; iz <= layers; iz++)
{
for(var i = 1; i <= 500; i++)
{
gotoframe(i,iz);
}
}
}
alert('done');
}

function gotoframe(framenumber, layer)
{
var wasKeyFrame = new Boolean(false);
for(var m = 0; m <= kFrameCount; m++)
{
if(KeyFrames[m] == currentFrameSelection + "," + currentLayerSelection){
wasKeyFrame = true
}
}
var aframe;
aframe = document.getElementById("frame" + currentFrameSelection + "layer" + currentLayerSelection);
if(wasKeyFrame == false){
aframe.style.color = "black";
aframe.style.backgroundColor="#E8E8E8";
}
if(wasKeyFrame == true){
aframe.style.color = "#000000";
aframe.style.backgroundColor="#0099CC";
}
currentFrameSelection = framenumber
currentLayerSelection = layer
var frame;
frame = document.getElementById("frame" + framenumber + "layer" + layer);
var isKeyFrame = new Boolean(false);
for(var m = 0; m <= kFrameCount; m++)
{
if(KeyFrames[m] == framenumber + "," + layer){
isKeyFrame = true
}
}
if(isKeyFrame == false){
frame.style.color = "#F2F2F2";
frame.style.backgroundColor="#00BFFF";
}
if(isKeyFrame == true){
frame.style.color = "#F2F2F2";
frame.style.backgroundColor="#3579DC";
}
}

function removeKeyframe(){
var wasKeyFrame = new Boolean(false);
for(var m = 0; m <= kFrameCount; m++)
{
if(KeyFrames[m] == currentFrameSelection + "," + currentLayerSelection){
wasKeyFrame = true
}
}
if(wasKeyFrame == false){
alert("Error: This isn't a Key Frame!")
}
if(wasKeyFrame == true){
if (confirm("Are You Sure you want to delete this frame?")) {
for(var m = 0; m <= kFrameCount; m++)
{
if(KeyFrames[m] == currentFrameSelection + "," + currentLayerSelection){
KeyFrames[m] = "0,0";
}
}
}
}
}
function ClearAllKeyframes(){
if (confirm("Are you sure you want to delete ALL the keyframes?")) {
for(var m = 0; m <= kFrameCount; m++)
{
KeyFrames[m] = "0,0";
}
alert("Done.");
if (confirm("Do you want to do a full timeline refresh to have all changes take effect?")) {
fullgotoframe();
}
}
}

function addLayer(){

var asdf;
layers = layers + 1;
asdf = "<div id='framesDiv'><table width='100%' id='framesTable' + layers cellspacing='0' border='1'>"
asdf = asdf + ("<tr>")
asdf = asdf + ("<td height='5' bgcolor='#F2F2F2' style='filter:alpha(opacity=75);-moz-opacity:.75;opacity:.75;'>Layer" + layers + "</td>");
for (var x = 1; x <= 500; x++)
{
asdf = asdf + ("<td id='frame" + x + "layer" + layers + "'height='5'");
asdf = asdf + ("bgcolor='#E8E8E8' style='filter:alpha(opacity=50);-moz-opacity:.50;opacity:.50;' onmousedown='gotoframe(" + x +", " + layers +");'><div id='FrameDiv' ");
asdf = asdf + ("onMouseover=\"ddrivetip('<b>frame:</b> "+x+"<br><b>layer:</b> "+layers+"');setTooltipData('"+x+"','"+layers+"')\" onMouseout=\"hideddrivetip()\">"+x+"</div></td>");
}
asdf = asdf + ("</tr></table></div>")
changeInnerHTML("frameContainer",asdf)
}


function setTooltipData(uFrame,uLayer){
var wasKeyFrame = new Boolean(false);
for(var m = 0; m <= kFrameCount; m++)
{
if(KeyFrames[m] == uFrame + "," + uLayer){
wasKeyFrame = true
}
}
var ziskeyframe = 'false';
ziskeyframe = 'false';
if(wasKeyFrame == true){
ziskeyframe = 'true';
}
var zisselected = 'false';
if (currentFrameSelection == uFrame){
if (currentLayerSelection == uLayer){
zisselected = 'true';
}
}
changeInnerHTML ('dhtmltooltip', '<br /><b>keyframe:</b> ' + ziskeyframe)
changeInnerHTML ('dhtmltooltip', '<br /><b>selected:</b> ' + zisselected)
}


function changeInnerHTML (elm, txt) {
if(document.getElementById) {
var el = document.getElementById(elm);
el.innerHTML = el.innerHTML + txt;
return true;
}
return false;
}
function overwriteInnerHTML (elm, txt) {
if(document.getElementById) {
var el = document.getElementById(elm);
el.innerHTML = txt;
return true;
}
return false;
}

function createjsDOMenu() {
timelineContextMenu = new jsDOMenu(170);
with (timelineContextMenu) {
addMenuItem(new menuItem("Add New Layer", "", "javascript:addLayer();"));
addMenuItem(new menuItem("Convert To Keyframe", "", "javascript:toKeyframe();"));
addMenuItem(new menuItem("Remove Keyframe", "", "javascript:removeKeyframe();"));
}
filter = new Array("*.frameContainer","td.*","div.frameDiv","img.frameIMG");
timelineContextMenu.setNoneExceptFilter(filter);
setPopUpMenu(timelineContextMenu);
activatePopUpMenuBy(1, 2);
}

function zaddLayer(){
addLayer();
overwriteInnerHTML('InitStatus','Finishing Timeline');
setTimeout("waitHideLoader();",1000);
}

function waitHideLoader(){
setTimeout("HideLoader();",1000);
overwriteInnerHTML('InitStatus','Rendering');
}

function zinit(){
setTimeout("zaddLayer();",1000);
overwriteInnerHTML('InitStatus','Generating Timeline');
initjsDOMenu();
}
