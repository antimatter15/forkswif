Ok, ok... how is this...
Code:
helicopter::onEnterFrame():
// Your movement code stays here.

// Rotate helicopter
x1=_root._xmouse;
y1=_root._ymouse;
x2=this.pivot._x+this._x+200;
y2=this.pivot._y+this._y+200;
this._rotation=(Math.atan((y1-y2)/(x1-x2))*180/Math.PI)+(x1<x2?180:0);

// Reduce time until able to fire
if(holdFire>0){
   holdFire--;
}

// If trying to fire, and able to fire, then fire
if(bulletCount&&!holdFire){
   _root.attachMovie('bullet','bullet',bullets);
   x=this._x;
   y=this._y;
   angle=this._rotation;
   _root.bullet.initialize(angle, x, y);
   _root.bullet._name="bullet"+bullets;
   bullets++;
   bulletCount--;
   // if you want for streams of bullets to take a longer time coming out of the gun, do a holdFire=<number of frames>; here
}

Code:
helicopter::onMouseDown():
if(!holdFire){
   bulletCount=5;
   holdFire=7; // how long to wait until the mouse can be pressed again (number of frames).  20 frames @ 3 frames per second = 20/3 ~=~ 7.
}

Code:
helicopter::onLoad():
bulletCount=0;
holdFire=0;
