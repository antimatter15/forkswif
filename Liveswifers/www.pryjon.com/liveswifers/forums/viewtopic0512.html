   $y1 = $_REQUEST['y1'];
   $y2 = $_REQUEST['y2'];
   $speed = $_REQUEST['speed'];

  $CharacterDepth = $swf->EasyPlaceObject($CharacterInfo["CharacterID"]);
  $charid = $CharacterInfo["CharacterID"];

  $y = 0;
  for ($x = 1; $x <= 500; $x++) {
  $y = $x * 20 * $speed; //add 1 px
  $swf->RemoveObjectFromLayer($charid, $chardepth);
  $CharacterInfo = $swf->DefineStraightLine($x1 + $y, $y1+ $y, $x2+ $y, $y2+$y, 100, false, 0, 0, 0, 0);

  }
