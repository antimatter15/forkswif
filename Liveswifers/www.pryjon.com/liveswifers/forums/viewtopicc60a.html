Replace the code that you have on the player with this:

Code:

if (key.isDOWN(Key.UP)) {
this._y -= 6;
}   
if (key.isDOWN(Key.DOWN)) {
this._y += 6;
}
if (key.isDOWN(Key.LEFT)) {
this._x -= 6;
}
if (key.isDOWN(Key.RIGHT)) {
this._x += 6;
}
if(this.hitTest(_root.wall1)){
   if(key.isDown(key.RIGHT)){
      this._x -= 6;
   }
      if(key.isDown(key.LEFT)){
         this._x += 6;
   }
      if(key.isDown(key.UP)){
         this._y += 6;
   }
      if(key.isDown(key.DOWN)){
         this._y -= 6;
   }
}
